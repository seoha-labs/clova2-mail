# Epic E — Send History (full body, 50-entry ring buffer) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Record every email send (success and failure) into a 50-entry ring buffer in `chrome.storage.local`, with full body capture, quota-safe truncation, and a popup UI to view / re-send / delete history entries.

**Architecture:** A pure, side-effect-free helper module (`src/shared/historyBuffer.ts`) owns all ring-buffer + quota logic so it is unit-testable under the existing `tests/**/*.test.ts` vitest glob. `src/shared/storage.ts` adds thin async accessors (`getSendHistory` / `appendSendHistory` / `deleteHistoryEntry` / `clearSendHistory`) that delegate to the helper. `src/background/index.ts` records a history entry on **both** the success and failure path of `SEND_EMAIL`, best-effort (a history write failure never fails the send). The popup gains a `SendHistory` component plus accessors; re-send pushes recipients + body into the content-script modal via a stored "pending re-send" payload picked up by the modal on mount.

**Tech Stack:** TypeScript, React 19, Vitest 4 (jsdom), `chrome.storage.local`. No new dependencies.

---

## Why a separate `historyBuffer.ts` helper

The vitest config (`vitest.config.ts`) only includes `tests/**/*.test.ts` — **not** `.tsx`. React component logic therefore cannot be unit-tested through the existing harness. To keep Epic E's quota / ring-buffer / truncation logic fully tested, all of that logic lives in a **pure** module (`src/shared/historyBuffer.ts`) that takes data in and returns data out (no `chrome` calls, no `Date.now()` inside). Determinism (id + timestamp) is injected by the caller, so tests pass fixed values directly and never need to stub globals for the helper. The storage accessors and the background handler are exercised by `.ts` integration tests that stub `chrome` + `Date.now()` + `Math.random()` exactly like the existing suites.

/** Per-entry cap on bodyHtml: 256 KB of UTF-8 bytes. */
export const MAX_BODY_BYTES = 256 * 1024;

/** Visible marker appended when a body is truncated (Korean UI). */
const TRUNCATION_MARKER = '\n\n<p>… [내용이 잘렸습니다]</p>';

const encoder = new TextEncoder();
const decoder = new TextDecoder();

export interface TruncateResult {
  readonly bodyHtml: string;
  readonly truncated: boolean;
}

/**
 * Caps bodyHtml at MAX_BODY_BYTES UTF-8 bytes. When over the cap, cuts at a
 * byte boundary that never splits a multibyte char (TextDecoder with
 * { stream:false } drops a trailing partial sequence rather than emitting
 * U+FFFD when we slice on a clean boundary) and appends a visible marker.
 */
export function truncateBody(bodyHtml: string): TruncateResult {
  const bytes = encoder.encode(bodyHtml);
  if (bytes.length <= MAX_BODY_BYTES) {
    return { bodyHtml, truncated: false };
  }
  const markerBytes = encoder.encode(TRUNCATION_MARKER).length;
  const budget = MAX_BODY_BYTES - markerBytes;
  // Decode a prefix; `decoder.decode` on a non-streaming call replaces a
  // trailing partial multibyte sequence, so trim back to a clean boundary.
  let cut = budget;
  let head = decoder.decode(bytes.subarray(0, cut));
  // Drop a trailing replacement char if the slice landed mid-codepoint.
  if (head.endsWith('�')) {
    head = head.slice(0, -1);
  }
  return { bodyHtml: head + TRUNCATION_MARKER, truncated: true };
}
```

export interface MakeSentEmailInput {
  readonly id: string;
  readonly sentAt: number;
  readonly to: readonly string[];
  readonly cc?: readonly string[];
  readonly bcc?: readonly string[];
  readonly subject: string;
  readonly bodyHtml: string;

  readonly success: boolean;
  readonly error?: string;

/**
 * Builds an immutable SentEmail. cc/bcc default to [] when absent (Epic A may
 * not be present). Determinism: id + sentAt are injected by the caller, never
 * generated here, so tests pass fixed values.
 */
export function makeSentEmail(input: MakeSentEmailInput): SentEmail {
  const { bodyHtml, truncated } = truncateBody(input.bodyHtml);
  const base: SentEmail = {
    id: input.id,
    sentAt: input.sentAt,
    to: [...input.to],
    cc: input.cc ? [...input.cc] : [],
    bcc: input.bcc ? [...input.bcc] : [],
    subject: input.subject,
    bodyHtml,
    mode: input.mode,
    success: input.success,
    truncated,
  };
  return input.error === undefined ? base : { ...base, error: input.error };
}

/** Ring-buffer length cap. */
export const MAX_HISTORY_ENTRIES = 50;

/**
 * Total serialized-history byte budget. chrome.storage.local allows ~5MB
 * across all keys; reserve a conservative slice for the history key so other
 * keys (recipients, templates) keep headroom.
 */
export const MAX_TOTAL_BYTES = 4 * 1024 * 1024;

const totalEncoder = new TextEncoder();

function serializedBytes(history: readonly SentEmail[]): number {
  return totalEncoder.encode(JSON.stringify(history)).length;
}

/**
 * Returns a NEW history array with `entry` prepended (newest first), trimmed to
 * MAX_HISTORY_ENTRIES, then trimmed further by dropping the oldest (tail) until
 * the serialized size fits MAX_TOTAL_BYTES. The newest entry is never dropped.
 */
export function appendToBuffer(
  history: readonly SentEmail[],
  entry: SentEmail,
): readonly SentEmail[] {
  let next: SentEmail[] = [entry, ...history].slice(0, MAX_HISTORY_ENTRIES);
  // Drop oldest until it fits, but always keep at least the newest entry.
  while (next.length > 1 && serializedBytes(next) > MAX_TOTAL_BYTES) {
    next = next.slice(0, -1);
  }
  return next;
}
```

export async function getSendHistory(): Promise<readonly SentEmail[]> {
  return (await get('sendHistory')) ?? [];

/** Payload for a new history entry; id + sentAt are generated here. */
export type AppendHistoryInput = Omit<MakeSentEmailInput, 'id' | 'sentAt'>;

/**
 * Builds a SentEmail (generating id + sentAt) and writes the trimmed,
 * quota-guarded history back. Returns the created entry. In tests, stub
 * Date.now() and Math.random() for determinism.
 */
export async function appendSendHistory(input: AppendHistoryInput): Promise<SentEmail> {
  const sentAt = Date.now();
  const id = `${sentAt}-${Math.random().toString(36).slice(2, 10)}`;
  const entry = makeSentEmail({ ...input, id, sentAt });
  const current = await getSendHistory();
  const next = appendToBuffer(current, entry);
  await set('sendHistory', next as StorageSchema['sendHistory']);
  return entry;
}

export async function deleteHistoryEntry(id: string): Promise<void> {
